32 void DisplayResultOrError(std::future<ResultType>& result_future, bool& calculation_started, std::string& result, std::string& error_str) {
33  if (calculation_started) {
34  error_str.clear();
35  if (!result_future.valid()) {
36  error_str = "Error starting the computation";
37  calculation_started = false;
38  } else if (result_future.wait_for(std::chrono::seconds(0)) == std::future_status::ready) {
39  try {
40  if constexpr (std::is_same_v<ResultType, LongNatural> || std::is_same_v<ResultType, LongInteger> || std::is_same_v<ResultType, Polynomial>) {
41  result = result_future.get().toString();
42  } else if constexpr (std::is_same_v<ResultType, LongRational>) {
43  LongRational res = result_future.get();
44  result = '\n' + res.getNumerator().toString() + "/\n" + res.getDenominator().toString();
45  } else if constexpr (std::is_same_v<ResultType, int>) {
46  result = std::to_string(result_future.get());
47  } else if constexpr (std::is_same_v<ResultType, bool>) {
48  result = result_future.get() ? "true" : "false";
49  } else {
50  static_assert(std::is_same_v<ResultType, LongNatural> || std::is_same_v<ResultType, int> || std::is_same_v<ResultType, bool>,
51  "Unsupported result type");
52  result = "Unsupported result type";
53  }
54  //
55  } catch (const std::invalid_argument& e) {
56  error_str = "Invalid input: " + std::string(e.what());
57  } catch (const std::logic_error& e) {
58  error_str = "Error during computations: " + std::string(e.what());
59  } catch (const std::bad_alloc& e) {
60  error_str = "Error allocating memory: " + std::string(e.what());
61  } catch (const std::exception& e) {
62  error_str = "Unknown error during computations: " + std::string(e.what());
63  }
64  calculation_started = false;
65  } else {
66  ImGui::Text("Calculating...");
67  }
68  } else {
69  ImGui::BeginChild("Result", ImVec2(0, ImGui::GetTextLineHeightWithSpacing() * 5), true, ImGuiWindowFlags_HorizontalScrollbar);
70  if (!result.empty()) ImGui::Text("Result: %s", result.c_str());
71  ImGui::EndChild();
72  }
73  if (!result.empty() && ImGui::Button("Copy to clipboard")) {
74  ImGui::SetClipboardText(result.c_str());
75  }
76  if (!error_str.empty()) ImGui::TextColored(ImVec4(1.0f, 0.0f, 0.0f, 1.0f), "%s", error_str.c_str());
77 }
