40std::string MapToPolyStr(const std::map<LongNatural, LongRational>& terms) {
41 if (terms.empty()) {
42 return "0";
43 }
44 std::string result = "";
45 // добавляем в строку коэффиценты с наибольшей степени
46 for (auto it = terms.rbegin(); it != terms.rend(); ++it) {
47 auto [deg, coef] = *it;
48 if (deg == LongNatural("0")) {
49 result += coef.toString() + " + ";
50
51 } else {
52 result += "(" + coef.toString() + ")" + "x^" + deg.toString() + " + ";
53 }
54 // если строка получается больше 10000 символово обрезаем строку
55 if (result.size() > 10000) {
56 return result.substr(0, result.size() - 3) + "..." + "и еще " + std::to_string(terms.size() - std::distance(terms.rbegin(), it)) +
57 " ненулевых членов";
58 }
59 }
60 return result.substr(0, result.size() - 3);
61}
